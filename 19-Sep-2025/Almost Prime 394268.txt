bool p(int n){
    set<int> s;
    for (int i = 2; i*i <= n; i++)
    {
        while (n%i==0){
            s.insert(i);
            n/=i;
        }
    }
    if(n>1)s.insert(n);
    return s.size()==2;
}
vector<bool> linearSieve(int N)
{
    vector<bool>isPrime(N,1);vector<int>primes;
    isPrime[0]=isPrime[1]=0;
    for(ll i=2;i<N;i++)
    {
        if(isPrime[i]) primes.push_back(i);
        for(auto it:primes)
        {
            if(i*it>=N) break;
            isPrime[i*it]=0;
            if(i%it==0) break;
        }
    }
    return isPrime;
}

void solve()
{
    int n;
    cin>>n;
    vector<bool> q = linearSieve(n+1);
    int sum=0;
    for (int i = 1; i <= n; i++)
    {
        if(!q[i])
        sum+=p(i);
    }
    cout<<sum;
    
}
