bool Knowbit(ll n,int i)
{
    return (n>>i)&1;

{
    return n|(1<<i);
}
ll Resetbit(ll n,int i)
{
    return n&(~(1<<i));
}
ll flip(ll n,int i)
{
    return n^(1<<i);
}
bool isPowerOfTwo(ll n)
{
    if(n==0)return 0;
    return !(n&(n-1));
}

int sum(int t,int x){
    while (t<=x)
    {
        t*=2;
    }
    t/=2;
    bitset<32> k = x-t;
    return k.count();
}

void solve()
{
    int x;
    cin>>x;
    bitset<32> k =x;
    int i=0;
    while (true)
    {
        if(k[i]==1)break;
        i++;
    }
    if(i==0){
        if(k.count()==1)cout<<3<<endl;
        else cout<<1<<endl;
        return;
    }
    int h = 1<<(i);

    if(k.count()==1){
        h++;
    }

    cout<<h<<endl;    
}
