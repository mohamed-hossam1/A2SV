int x, k;
bool p(int m)
{
    int sum = 0;
    int t = 1;
    while (sum < x)
    {
        if (m / t == 0)
            break;
        sum += m / t;
        t *= k;
    }
    return sum >= x;
}
void solve()
{

    int x;
    cin >> x;
    int arr[x];
    for (int i = 0; i < x; i++)
    {
        cin >> arr[i];
    }
    sort(arr,arr+x);
    int q;
    cin >> q;
    while (q--)
    {
        int w;
        cin>>w;
        int l = 0, r = x - 1, ans = -1;
        while (l <= r)
        {
            int mid = (l + r) / 2;
            if (arr[mid] <= w)
            {
                ans = mid;
                l = mid + 1;
            }
            else
            {
                r = mid - 1;
            }
        }
        cout << ans + 1<<endl;
    }
}
