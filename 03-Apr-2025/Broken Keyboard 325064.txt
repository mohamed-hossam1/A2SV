void solve()
{
    int x,d;
    cin>>x>>d;
    pair<int,int> p[x];
    for (int i = 0; i < x; i++)
    {
        cin>>p[i].first>>p[i].second;
    }
    sort(p,p+x);
    int m = p[0].first;
    int sum=p[0].second,k=-1;
    int l =0;
    for (int i = 1; i < x; i++)
    {
        if(p[i].first-m<d){
            sum+=p[i].second;
        }
        else{
            k = max(k,sum);
            
            sum-=p[l].second;
            l++;
            m = p[l].first;
            i--;


        }
    }
    
    cout<<max(k,sum);
    
    
    

    
    
}
