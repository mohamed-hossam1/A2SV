# Problem:  Jellyfish and Mex - https://codeforces.com/problemset/problem/1875/D

#include <bits/stdc++.h>
using namespace std;

void solve() {

    int n, x;
    cin >> n;
    int dp[n + 1];
    memset(dp, 0x3f, sizeof(dp));
    map<int, int> freq;
    for (int i = 0; i < n; i++) {
        cin >> x;
        freq[x]++;
    }
    dp[n] = 0;
    for (int i = n; i >= 0; i--) {
        for (int j = i + 1; j <= n; j++) {
            dp[i] = min(dp[i], dp[j] + max(freq[i] - 1, 0) * j + (freq[i] ? i : 0));
        }
    }
    cout << dp[0];

}

int main() {

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) {
        solve();
        if (t) cout << "\n";
    }

    return 0;
}
